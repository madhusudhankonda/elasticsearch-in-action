# Match queries

##Match all query
GET books/_search
{
  "query": {
    "match_all": {}
  }
}

GET books/_search
{
  "query": {
    "match": {
      "author": "Cay"
    }
  }
}

# Match query searching Java titles
GET books/_search
{
  "query": {
    "match": {
      "title": "Java"
    }
  }
}

# Match query searching for a set of words (like a sentence)
GET books/_search
{
  "query": {
    "match": {
      "title": {
        "query": "Java Complete Guide"
      }
    }
  },
  "highlight": {
    "fields": {
      "title": {}
    }
  }
}
