# Finding the number of documents using value_count metric
GET tv_sales/_search
{
  "size": 0, 
  "aggs": {
    "total-values": {
      "value_count": {
        "field": "brand.keyword"
      }
    }
  }
}

# Finding the average price of the mobile
GET tv_sales/_search
{
  "size": 0, 
  "aggs": {
    "tv_average_price": {
      "avg": {
        "field": "price_gbp"
      }
    }
  }
}
